using UnityEngine;


public class FlashingLightsMgr : MonoBehaviour
{
   public FlashingLightMgr flashingLightLeft;
   public FlashingLightMgr flashingLightRight;
   public FlashingLightMgr flashingLightReverse;
   readonly float intervalFlashing = .25f;
   private bool ynFlashingOn;


   private void Start()
   {
       StartFlashing();
   }
  
   void StartFlashing()
   {
       ynFlashingOn = true;
       Flash();
   }
  
   void Flash()
   {
       if (ynFlashingOn)
       {
           ColorSignal(flashingLightLeft, Color.red);
           ColorSignal(flashingLightRight, Color.red);
           ColorSignal(flashingLightReverse, Color.red);
       }
       else
       {
           ColorSignal(flashingLightLeft, Color.darkRed);
           ColorSignal(flashingLightRight, Color.darkRed);
           ColorSignal(flashingLightReverse, Color.darkRed);
       }
       ynFlashingOn = !ynFlashingOn;
->     Invoke(nameof(Flash), intervalFlashing); 
   }


   void ColorSignal(FlashingLightMgr signal, Color color)
   {
       signal.SetColor(color);
   }
}

CLASS 7 TOLERANCE
Tolerance is determining how much is close enough.  Navigation checks when the robot has arrived at the next waypoint. 

ARRIVING AT A WAYPOINT
The target distance is calculated 
The robot moves a delta distance each frame. 
As the robot approaches the waypoint, the distance traveled gets larger.
When the distance traveled >= the target distance, the robot has arrived
The tolerance can range from 0 to the delta distance
0 <= tolerance < delta distance


CLASS 8 TIME CONSISTENCY
Time consistency becomes an issue when you run the same code on differently powered devices
Update() is run every frame
The frame rate is different for each device type
Knowing how long a frame takes solves this issue


TIMING CONSISTENCY
Desktop editors run at a higher frame rate than mobile or headset devices
Movement speed needs to be consistent across various device
Time.delta is our friend
Speed is Meters Per Second
Time.deltaTime is how long each frame update takes, 100 frames per second = .01 deltaTime
Time.deltaTime is calculated each frame (accurate)
Each update move = Speed x deltaTime
So if speed = 2 m/s, and deltaTime = .025 (40fps), move = 2 x .025 = .05m each frame

LaserMgr.cs
using UnityEngine;
using TMPro;


public class LaserMgr : MonoBehaviour
{
   public GameObject ray;
   public GameObject point;
   public float distCurrent;
   readonly float intervalCheck = .1f;
   readonly float rangeLaser = 2f;
   public GameObject sensor;
   Vector3 origin;
   public TMP_Text textInfo;
   readonly float thickness = .01f;
   RaycastHit hit;
   Camera cam;
   Renderer rend;


   void Start()
   {
       rend = sensor.GetComponent<Renderer>();
       cam = Camera.main;
       InvokeRepeating(nameof(CheckDistance), intervalCheck, intervalCheck);
   }


   void Update()
   {
       UpdateTextInfo();
   }


   void UpdateTextInfo()
   {
       textInfo.transform.LookAt(cam.transform.position);
       textInfo.transform.Rotate(0,180,0);
   }


   public float GetDistance()
   {
       return distCurrent;
   }


   void CheckDistance()
   {
       origin = sensor.transform.position;
       Vector3 direction = sensor.transform.forward;
       if (Physics.Raycast(origin, direction, out hit, rangeLaser))
       {
           point.transform.position = hit.point;
           distCurrent = hit.distance;
           AdjustRay();
           ColorSensor(Color.red);
           textInfo.text = distCurrent.ToString("F1");
       }
       else
       {
           distCurrent = -1;
           point.transform.position = origin;
           ray.transform.position = origin;
           ray.transform.localScale = Vector3.one * thickness;
           ColorSensor(Color.green);
           textInfo.text = "-1";
       }
   }


   void ColorSensor(Color color)
   {
       rend.material.color = color;
   }
  
   void AdjustRay()
   {
       ray.transform.position = (origin + point.transform.position) / 2;
       ray.transform.LookAt(point.transform.position);
       ray.transform.localScale = new (thickness, thickness, distCurrent);
   }
}

CLASS 11 NAVIGATION
NavigationMgr.cs
using System.Collections.Generic;
using UnityEngine;


public class NavigationMgr : MonoBehaviour
{
   public GameObject robot;
   public MotorsMgr motorsMgr;
   public GameObject waypointsParent;
   readonly List<WaypointMgr> waypoints = new();
   WaypointMgr waypointCurrent;
   private int nCurrent;
   readonly float toleranceNear = .05f;
  
   public Vector3 look;
   public Vector3 target;
   readonly float durationRotate = 3;
   private float fractionRotate;
   private float deltaFractionRotate;
   private bool ynRotate;
   Vector3 vecLookTarget;
   readonly float speedMove = .5f;
  
   private LeftRight leftRightCurrent;
   public bool ynPause;
   private Vector3 source;
   private float distSourceTarget;


   void Start()
   {
       LoadWaypoints();
       nCurrent = -1;
       Advance();
   }


   void LoadWaypoints()
   {
       for (int n = 0; n < waypointsParent.transform.childCount; n++)
       {
           WaypointMgr waypoint = waypointsParent.transform.GetChild(n).GetComponent<WaypointMgr>();
           waypoints.Add(waypoint);
       }
   }


   bool IsTooFar()
   {
       bool ynTooFar = false;
       float dist = Vector3.Distance(source, robot.transform.position);
       if (dist > distSourceTarget)
       {
           ynTooFar = true;
       }
       return ynTooFar;
   }


   void Advance()
   {
       nCurrent++;
       if (nCurrent >= waypoints.Count)
       {
           nCurrent = 0;
       }
       waypointCurrent = waypoints[nCurrent];
       HighlightWaypointCurrent();
       OrientRobot();
   }


   void HighlightWaypointCurrent()
   {
       for (int n = 0; n < waypoints.Count; n++)
       {
           WaypointMgr waypoint = waypoints[n];
           if (waypoint == waypointCurrent)
           {
               waypoint.Highlight(true);
           }
           else
           {
               waypoint.Highlight(false);
           }
       }
   }


   void OrientRobot()
   {
       source = robot.transform.position;
       look = robot.transform.TransformPoint(Vector3.forward);
       target = waypointCurrent.transform.position;
       vecLookTarget = target - look;
       fractionRotate = 0;
       deltaFractionRotate = Time.deltaTime / durationRotate;
       ynRotate = true;
       Vector3 posL = robot.transform.InverseTransformPoint(target);
       if (posL.x < 0)
       {
           leftRightCurrent = LeftRight.left;
       }
       else
       {
           leftRightCurrent = LeftRight.right;
       }
       distSourceTarget = Vector3.Distance(source, target);
   }


   void UpdateRotate()
   {
       if (!ynRotate) return;
       if (fractionRotate >= 1)
       {
           robot.transform.LookAt(target);
           ynRotate = false;
           return;
       }
       Vector3 pos = look + vecLookTarget * fractionRotate;
       robot.transform.LookAt(pos);
       motorsMgr.Turn(leftRightCurrent);
       fractionRotate += deltaFractionRotate;
   }


   void UpdateMove()
   {
       if (ynRotate) return;
       if (!ynPause)
       {
           motorsMgr.Forward();
           robot.transform.Translate(0, 0, speedMove * Time.deltaTime);
       }
   }
  
   void Update()
   {
       if (IsTooFar())
       {
           Advance();
       }
       UpdateRotate();
       UpdateMove();
   }
}


public enum LeftRight
{
   left,
   right
}

CLASS 14 PEOPLE
PeopleMgr.cs
using System.Collections.Generic;
using UnityEngine;


public class PeopleMgr : MonoBehaviour
{
   public GameObject robot;
   readonly float radiusFloorEdge = 5;
   readonly float radiusRobot = 1.4f;
   readonly float radiusPerson = .7f;
   readonly List<PersonMgr> people = new();
   readonly float speedMove = 1;
   readonly float speedMoveRotate = .2f;
  
   void Start()
   {
       LoadPeople();
   }


   void LoadPeople()
   {
       for (int n = 0; n < transform.childCount; n++)
       {
           PersonMgr person = transform.GetChild(n).GetComponent<PersonMgr>();
           people.Add(person);
       }
   }


   bool IsOut(PersonMgr person)
   {
       bool ynOut = false;
       float dist = Vector3.Distance(person.transform.position, transform.position);
       if (dist > radiusFloorEdge)
       {
           ynOut = true;
       }
       return ynOut;
   }


   bool IsNearRobot(PersonMgr person)
   {
       bool ynNear = false;
       float dist = Vector3.Distance(person.transform.position, robot.transform.position);
       if (dist < radiusRobot)
       {
           Vector3 posL = person.transform.InverseTransformPoint(robot.transform.position);
           if (posL.z > 0)
           {
               ynNear = true;
           }
       }
       return ynNear;
   }


   bool IsNearPerson(PersonMgr person)
   {
       bool ynNear = false;
       for (int n = 0; n < people.Count; n++)
       {
           PersonMgr person2 = people[n];
           float dist = Vector3.Distance(person.transform.position, person2.transform.position);
           if (person != person2 && dist < radiusPerson)
           {
               Vector3 posL = person.transform.InverseTransformPoint(person2.transform.position);
               if (posL.z > 0)
               {
                   ynNear = true;
                   break;
               }
           }
       }
       return ynNear;
   }


   void CheckPeople()
   {
       for (int n = 0; n < people.Count; n++)
       {
           PersonMgr person = people[n];
           CheckPerson(person);
       }
   }


   void CheckPerson(PersonMgr person)
   {
       if (IsOut(person) || IsNearRobot(person) || IsNearPerson(person))
       {
->         RotatePerson(person);
->         MovePerson(person, true);
       }
       else
       {
->         MovePerson(person, false);
       }
   }


   void RotatePerson(PersonMgr person)
   {
       person.Rotate();
   }


   void MovePerson(PersonMgr person, bool ynRotate)
   {
       if (ynRotate)
       {
           person.Move(speedMoveRotate);
       }
       else
       {
           person.Move(speedMove);
       }
   }


   void Update()
   {
       CheckPeople();
   }
}

CLASS 15 PERSON WALKING
PERSON WALKING
A sine curve is used to rotate both arms and both legs to simulate walking
PersonMgr.cs
using UnityEngine;
using Random = UnityEngine.Random;


public class PersonMgr : MonoBehaviour
{
   public GameObject armLeft;
   public GameObject armRight;
   public GameObject legLeft;
   public GameObject legRight;
   readonly float deltaRotateAngle = 2;
   private int side;
   private float angle;
   readonly float speedAngleDelta = 250;
   private float randArmLegs;


   private void Start()
   {
       side = 1;
       if (Random.Range(0, 2) == 0)
       {
           side = -1;
       }
       randArmLegs = Random.Range(-3f, 3);
   }


   public void Rotate()
   {
       transform.Rotate(0, deltaRotateAngle * side, 0);
   }


   public void Move(float dist)
   {
       float rand = Random.Range(.9f, 1.1f);
       transform.Translate(0, 0, dist * rand * Time.deltaTime);
   }


-> void UpdateArmsLegs()
   {
       float ang = 15 * Mathf.Cos(angle * Mathf.Deg2Rad) + randArmLegs;
       armLeft.transform.localEulerAngles = new Vector3(0, 0, ang);
       armRight.transform.localEulerAngles = new Vector3(0, 0, -ang);
       legLeft.transform.localEulerAngles = new Vector3(0, 0, -ang);
       legRight.transform.localEulerAngles = new Vector3(0, 0, ang);
   }
  
   void Update()
   {
       UpdateArmsLegs();
       angle += speedAngleDelta * Time.deltaTime;
   }
}
