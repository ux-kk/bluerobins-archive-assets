SQL #2 — published_at column + backfill + index + trigger
-- 1. Add column
ALTER TABLE projects ADD COLUMN IF NOT EXISTS published_at TIMESTAMPTZ;
-- 2. Backfill: for projects that are currently available, treat created_at as publish date.
-- For drafts (not available), leave published_at NULL so they're excluded from "recent" lists.
UPDATE projects
SET published_at = created_at
WHERE published_at IS NULL
  AND availability_status = 'available'
  AND deleted = false;
-- 3. Index for the sort
CREATE INDEX IF NOT EXISTS idx_projects_published_at
  ON projects(published_at DESC)
  WHERE deleted = false AND availability_status = 'available';
-- 4. Trigger: stamp published_at the first time availability_status flips to 'available'
CREATE OR REPLACE FUNCTION set_project_published_at()
RETURNS TRIGGER AS $$
BEGIN
  IF NEW.availability_status = 'available'
     AND (OLD.availability_status IS DISTINCT FROM 'available')
     AND NEW.published_at IS NULL THEN
    NEW.published_at := NOW();
  END IF;
  RETURN NEW;
END;
$$ LANGUAGE plpgsql;
DROP TRIGGER IF EXISTS trg_set_project_published_at ON projects;
CREATE TRIGGER trg_set_project_published_at
  BEFORE UPDATE OF availability_status ON projects
  FOR EACH ROW
  EXECUTE FUNCTION set_project_published_at();
-- Also handle INSERT case (project created directly as 'available')
CREATE OR REPLACE FUNCTION set_project_published_at_on_insert()
RETURNS TRIGGER AS $$
BEGIN
  IF NEW.availability_status = 'available' AND NEW.published_at IS NULL THEN
    NEW.published_at := NOW();
  END IF;
  RETURN NEW;
END;
$$ LANGUAGE plpgsql;
DROP TRIGGER IF EXISTS trg_set_project_published_at_insert ON projects;
CREATE TRIGGER trg_set_project_published_at_insert
  BEFORE INSERT ON projects
  FOR EACH ROW
  EXECUTE FUNCTION set_project_published_at_on_insert();
-- 5. Verify
SELECT
  COUNT(*) FILTER (WHERE published_at IS NOT NULL) AS with_published_at,
  COUNT(*) FILTER (WHERE published_at IS NULL AND availability_status = 'available' AND deleted = false) AS missing_but_available,

FROM projects;
SELECT id, title, availability_status, created_at::date, published_at::date
FROM projects
WHERE deleted = false AND availability_status = 'available'
ORDER BY published_at DESC NULLS LAST
LIMIT 10;
